<script>
function submitForm() {
    // Get form data
    var formData = $("#projectForm").serialize();

    // Ajax request
    $.ajax({
        type: "POST",
        url: "process.php",
        data: formData,
        success: function(response) {
            // Handle success, if needed
            alert(response);
            window.location.href = "main.php";
        },
        error: function(error) {
            // Handle error, if needed
            alert("Error: " + error.statusText);
        }
    });
}
